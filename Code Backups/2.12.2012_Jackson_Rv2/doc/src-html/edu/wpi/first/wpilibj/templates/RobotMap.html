003/**
004 * The RobotMap is a mapping from the ports sensors and actuators are wired into
005 * to a variable name. This provides flexibility changing wiring, makes checking
006 * the wiring easier and significantly reduces the number of magic numbers
007 * floating around.
008 * 
009 * 
010 * Any changes to this map should be confirmed by a leader on the programming or
011 * electrical team!!!
012 */
013
014public class RobotMap {
015    //Modes
016    public static final boolean IS_REAL_BOT          = false;
017    public static final boolean DOES_USE_AUTO_RELOAD = false;
018    
019    //Sidecar: PWM Out
020    public static final int RIGHT_MOTORS  = 1;
021    public static final int LEFT_MOTORS   = 2;
022    public static final int ARM_MOTORS    = 3;
023    public static final int KICKER_MOTOR  = 4;
024    
025    
026    //Sidecar: Digital IO
027    public static final int CHASSIS_ENCODER_R_A        = 1;
028    public static final int CHASSIS_ENCODER_R_B        = 2;
029    public static final int CHASSIS_ENCODER_L_A        = 3;
030    public static final int CHASSIS_ENCODER_L_B        = 4;
031    public static final int PRESSURE_SWITCH            = 10;
032    public static final int KICKER_RETRACTED_SWITCH    = 13;
033    public static final int KICKER_MOTOR_RETURN_SWITCH = 14;
034    
035    //Sidecar: Relay
036    public static final int COMPRESSOR   = 1;
037    public static final int CAMERA_LIGHT = 4;
038    
039    
040    //Solenoid Breakout
041    public static final int TRANSMISSION_SOLENOID_LOW  = 1; //Pistons are extended
042    public static final int TRANSMISSION_SOLENOID_HIGH = 2; //Pistons are retracted
043    public static final int TABLE_TILT_UP              = 3; //Piston is extended
044    public static final int TABLE_TILD_DOWN            = 4; //Piston is retracted
045    public static final int KICKER_LATCH_RELEASE       = 5; //Piston is extended, this shoots the ball
046    public static final int KICKER_LATCH_RELOAD        = 6; //Piston is retracted, returning the latch to its normal state
047
048    
049    public static final int STICK_ONE = 1;
050    
051    //xbox controller button map
052    public static final int BUTTON_A          = 1;
053    public static final int BUTTON_RIGHT_TRIG = -3;
054    
055    
056    //Component Specifications
057    public static final double DRIVE_WHEEL_DIAMETER_FEET  = 0.5;
058    public static final double MOTOR_TO_WHEEL_GEAR_RATIO  = 12/30;
059    public static final int ENCODER_PULSES_PER_REVOLUTION = 360;
060    
061    //misc
062    public static final double KICKER_RELOAD_DELAY_SECONDS = 2.0;
063    
064    /*
065     * 1 - LeftX
066    2 - LeftY
067    3 - Triggers (Each trigger = 0 to 1, axis value = right - left)
068    4 - RightX
069    5 - RightY
070    6 - DPad Left/Right
071    * 
072    Button mapping matches Windows Control Panel>Game Pads display
073     */
074    
075}
